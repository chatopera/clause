13 namespace fs = boost::filesystem;

15 void copyDirectoryRecursively(const fs::path& sourceDir,
16  const fs::path& destinationDir) {
17  if (!fs::exists(sourceDir) || !fs::is_directory(sourceDir)) {
18  throw std::runtime_error("Source directory " + sourceDir.string() + " does not exist or is not a directory");
19  }
20 
21  if (fs::exists(destinationDir)) {
22  throw std::runtime_error("Destination directory " + destinationDir.string() + " already exists");
23  }
24 
25  if (!fs::create_directory(destinationDir)) {
26  throw std::runtime_error("Cannot create destination directory " + destinationDir.string());
27  }
28 
29  for (const auto& dirEnt : fs::recursive_directory_iterator{sourceDir}) {
30  const auto& path = dirEnt.path();
31  auto relativePathStr = path.string();
32  boost::replace_first(relativePathStr, sourceDir.string(), "");
33  fs::copy(path, destinationDir / relativePathStr);
34  }
35 }
36 
37 void copyDirectoryRecursively(const string& sourceDir, const string& destinationDir) {
38  fs::path src(sourceDir);
39  fs::path dest(destinationDir);
40  copyDirectoryRecursively(src, dest);
41 };
