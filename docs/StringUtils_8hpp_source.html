55  terms.clear();

57

60  terms.push_back(query.substr(i, 6));
61  i += 6;
62  sz++;
63  continue;
64  }
65

67  terms.push_back(query.substr(i, 5));
68  i += 5;
69  sz++;
70  continue;
71  }
72

74  terms.push_back(query.substr(i, 4));
75  i += 4;
76  sz++;
77  continue;
78  }
79 
80  if ((query[i] & 0xE0) == 0xE0 && i + 3 <= query.size()) {
81  terms.push_back(query.substr(i, 3));
82  i += 3;
83  sz++;
84  continue;
85  }
86

88  terms.push_back(query.substr(i, 2));
89  i += 2;
90  sz++;
91  continue;
92  }
93

216 template<typename STR>
217 TrimPositions TrimStringT(const STR& input,
218  const typename STR::value_type trim_chars[],
219  TrimPositions positions,
220  STR* output) {
221 
222  const typename STR::size_type last_char = input.length() - 1;
223  const typename STR::size_type first_good_char = (positions & TRIM_LEADING) ?
224  input.find_first_not_of(trim_chars) : 0;
225  const typename STR::size_type last_good_char = (positions & TRIM_TRAILING) ?
226  input.find_last_not_of(trim_chars) : last_char;
227 
228  if (input.empty() ||
229  (first_good_char == STR::npos) || (last_good_char == STR::npos)) {
230  bool input_was_empty = input.empty();
231  output->clear();
232  return input_was_empty ? TRIM_NONE : positions;
233  }
234 
235  *output =
236  input.substr(first_good_char, last_good_char - first_good_char + 1);
237 
238  return static_cast<TrimPositions>(
239  ((first_good_char == 0) ? TRIM_NONE : TRIM_LEADING) |
240  ((last_good_char == last_char) ? TRIM_NONE : TRIM_TRAILING));
241 }

380 template <class Uint16Container>
381 bool Utf8ToUnicode(const char * const str, size_t len, Uint16Container& vec) {
382  if(!str) {
383  return false;
384  }
385 
386  char ch1, ch2;
387  uint16_t tmp;
388  vec.clear();
389 
390  for(size_t i = 0; i < len;) {
391  if(!(str[i] & 0x80)) { // 0xxxxxxx
392  vec.push_back(str[i]);
393  i++;
394  } else if ((uint8_t)str[i] <= 0xdf && i + 1 < len) { // 110xxxxxx
395  ch1 = (str[i] >> 2) & 0x07;
396  ch2 = (str[i + 1] & 0x3f) | ((str[i] & 0x03) << 6 );
397  tmp = (((uint16_t(ch1) & 0x00ff ) << 8) | (uint16_t(ch2) & 0x00ff));
398  vec.push_back(tmp);
399  i += 2;
400  } else if((uint8_t)str[i] <= 0xef && i + 2 < len) {
401  ch1 = ((uint8_t)str[i] << 4) | ((str[i + 1] >> 2) & 0x0f );
402  ch2 = (((uint8_t)str[i + 1] << 6) & 0xc0) | (str[i + 2] & 0x3f);
403  tmp = (((uint16_t(ch1) & 0x00ff ) << 8) | (uint16_t(ch2) & 0x00ff));
404  vec.push_back(tmp);
405  i += 3;
406  } else {
407  return false;
408  }
409  }
410 
411  return true;
412 }

419 template <class Uint32Container>
420 bool Utf8ToUnicode32(const string& str, Uint32Container& vec) {
421  uint32_t tmp;
422  vec.clear();
423 
424  for(size_t i = 0; i < str.size();) {
425  if(!(str[i] & 0x80)) { // 0xxxxxxx
426  // 7bit, total 7bit
427  tmp = (uint8_t)(str[i]) & 0x7f;
428  i++;
429  } else if ((uint8_t)str[i] <= 0xdf && i + 1 < str.size()) { // 110xxxxxx
430  // 5bit, total 5bit
431  tmp = (uint8_t)(str[i]) & 0x1f;
432 
433  // 6bit, total 11bit
434  tmp <<= 6;
435  tmp |= (uint8_t)(str[i + 1]) & 0x3f;
436  i += 2;
437  } else if((uint8_t)str[i] <= 0xef && i + 2 < str.size()) { // 1110xxxxxx
438  // 4bit, total 4bit
439  tmp = (uint8_t)(str[i]) & 0x0f;
440 
441  // 6bit, total 10bit
442  tmp <<= 6;
443  tmp |= (uint8_t)(str[i + 1]) & 0x3f;
444 
445  // 6bit, total 16bit
446  tmp <<= 6;
447  tmp |= (uint8_t)(str[i + 2]) & 0x3f;
448 
449  i += 3;
450  } else if((uint8_t)str[i] <= 0xf7 && i + 3 < str.size()) { // 11110xxxx
451  // 3bit, total 3bit
452  tmp = (uint8_t)(str[i]) & 0x07;
453 
454  // 6bit, total 9bit
455  tmp <<= 6;
456  tmp |= (uint8_t)(str[i + 1]) & 0x3f;
457 
458  // 6bit, total 15bit
459  tmp <<= 6;
460  tmp |= (uint8_t)(str[i + 2]) & 0x3f;
461 
462  // 6bit, total 21bit
463  tmp <<= 6;
464  tmp |= (uint8_t)(str[i + 3]) & 0x3f;
465 
466  i += 4;
467  } else {
468  return false;
469  }
470 
471  vec.push_back(tmp);
472  }
473 
474  return true;
475 }
