48  void rank(WordMap &ws, size_t rankTime = 10) {
49  WordMap outSum;
50  Score wsdef, min_rank, max_rank;
51 
52  if( graph.size() == 0)
53  return;
54 
55  wsdef = 1.0 / graph.size();
56 
57  for(Graph::iterator edges = graph.begin(); edges != graph.end(); ++edges) {
58  // edges->first start节点；edge->first end节点；edge->second 权重
59  ws[edges->first].word = edges->first;
60  ws[edges->first].weight = wsdef;
61  outSum[edges->first].weight = 0;
62 
63  for(Edges::iterator edge = edges->second.begin(); edge != edges->second.end(); ++edge) {
64  outSum[edges->first].weight += edge->second;
65  }
66  }
67 
68  //sort(nodeSet.begin(),nodeSet.end()); 是否需要排序?
69  for( size_t i = 0; i < rankTime; i++ ) {
70  for(NodeSet::iterator node = nodeSet.begin(); node != nodeSet.end(); node++ ) {
71  double s = 0;
72 
73  for( Edges::iterator edge = graph[*node].begin(); edge != graph[*node].end(); edge++ )
74  // edge->first end节点；edge->second 权重
75  s += edge->second / outSum[edge->first].weight * ws[edge->first].weight;
76 
77  ws[*node].weight = (1 - d) + d * s;
78  }
79  }
80 
81  min_rank = max_rank = ws.begin()->second.weight;
82 
83  for(WordMap::iterator i = ws.begin(); i != ws.end(); i ++) {
84  if( i->second.weight < min_rank ) {
85  min_rank = i->second.weight;
86  }
87 
88  if( i->second.weight > max_rank ) {
89  max_rank = i->second.weight;
90  }
91  }
92 
93  for(WordMap::iterator i = ws.begin(); i != ws.end(); i ++) {
94  ws[i->first].weight = (i->second.weight - min_rank / 10.0) / (max_rank - min_rank / 10.0);
95  }
96  }
